# Tsup

## Install package
```bash
yarn add --dev tsup
```

## Configure
```bash
nano tsup.config.ts
```

```bash
import { defineConfig } from "tsup";

export default defineConfig({
    entry: ["src/myCalc.ts"],
    format: ["cjs", "esm"],
    // target: "es2022",
    dts: true,
});
```

## Scripts

```bash
{
    ...
    "scripts": {
        "build": "tsup"
    },
    ...
}
```
